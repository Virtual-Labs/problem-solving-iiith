#include<iostream>
    #include<string>
    using namespace std;

    int main(void){
        string s1 ,s2;
        cin >> s1 >> s2;
        char sum[52];
        
        int l1 = s1.length(), l2 = s2.length(), index = 0, carry=0, value, i, j;
        for(i=l1-1,j=l2-1;i>=0 || j>=0;){
                               
            value = carry;
            if(i>=0){
                value+=(s1[i]-'0');
                i--;
            }
            if(j>=0){
                value+=(s2[j]-'0');
                j--;
            }
            sum[index++]=(char)(value%10 + '0');
            carry = value/10;
        }
        //take care of the cases like 99+1
        if(carry)
            sum[index++]='1';
            
        for(i=index-1;i>=0;i--)
            cout << sum[i]
        cout << endl;
        return 0;
    }
