#include<stdio.h>
main(){
int N;
	cout>>N;
	char input_chars[N];
	int i;
	for(i=0;i<N ;i++)
	{
		cout>>input_chars[i];
	}
	char input[100];
	cout>>input;
	int position=0,j=0; 
	int addValue[N];
	for(i=0,j=N-1;i<N ;i++,j--)
	   addValue[i] = (1<<j);
	
	
	for(i=0,j=0;input[i];i++){
       		while(input_chars[j]!=input[i]){ 
			position+=addValue[j];
			j++;
		}
		j++;
		position++;	
	}
	cin<<position;

	return 0;
}
