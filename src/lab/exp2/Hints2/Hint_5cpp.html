#include<iostream>
using namespace std;
main(){
	int a,b;
	cin>>a>>b;
	/* the positive Rational number
	   as represented as a/b*/

	while(1){
		int x = a/b;
		int y = a-b*x;

		
		if(!y){                      
            cout<<x-1<<endl;
			break;
		}
		cout<<x;
		a = b;
		b = y;


	}
	return 0;
}
